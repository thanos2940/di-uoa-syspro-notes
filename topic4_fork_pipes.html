fork() — Δημιουργία Νέας Διεργασίας

Η fork() δημιουργεί ένα πανομοιότυπο αντίγραφο της τρέχουσας διεργασίας. Μετά την fork υπάρχουν δύο διεργασίες που εκτελούν τον ίδιο κώδικα — διαφέρουν μόνο στην τιμή που επέστρεψε η fork().

#include <unistd.h> #include <sys/types.h> pid_t pid = fork(); if (pid == -1) { perror("fork"); exit(1); // ↑ η fork() απέτυχε (σπάνιο) } else if (pid == 0) { // ΕΔΩ ΕΙΜΑΙ ΤΟ CHILD // pid == 0 σημαίνει "είμαι το παιδί" printf("Child, PID=%d\n", getpid()); exit(0); } else { // ΕΔΩ ΕΙΜΑΙ ΤΟ PARENT // pid == PID του child που μόλις φτιάξαμε printf("Parent, child PID=%d\n", pid); wait(NULL); // περιμένω να τελειώσει το child }
PARENT
καλεί fork()
↓ fork() ↓
PARENT
pid = child_PID
(θετικός αριθμός)
CHILD
pid = 0
("είμαι το παιδί")
ERROR
pid = -1
(μόνο στον parent)
🧠
Tip — τιμές επιστροφής
Parent παίρνει το PID του child (>0) — λογικό, γιατί ο parent θέλει να ξέρει ποιο παιδί έφτιαξε.
Child παίρνει 0 — "είμαι το παιδί, δεν γνωρίζω το PID μου ακόμα". Το δικό του PID βρίσκεται με getpid().
-1 = αποτυχία (π.χ. εξαντλήθηκε ο πίνακας διεργασιών).
> 0
Είμαι ο PARENT — αυτή είναι η τιμή = PID του child
= 0
Είμαι το CHILD — "0" σημαίνει "είμαι το παιδί"
= -1
ΑΠΟΤΥΧΙΑ — ελέγξε errno (EAGAIN, ENOMEM)
Τι κληρονομεί το child;
✅ Κληρονομεί
Αντίγραφο μνήμης (code, stack, heap) · Αντίγραφο FD table (ανοιχτά αρχεία, pipes) · Signal handlers · Environment variables · Τρέχων κατάλογος
✗ ΔΕΝ κληρονομεί
Pending signals · Memory locks · File locks · Timers (alarm) · PID (έχει νέο PID)
🧠
Tip — Copy-on-Write
Το child δεν παίρνει αμέσως αντίγραφο μνήμης — μοιράζονται την ίδια φυσική μνήμη ώσπου κάποιος γράψει σε κάτι. Αυτό κάνει τη fork() γρήγορη ακόμα και για μεγάλες διεργασίες.

exec() family — Εκτέλεση Νέου Προγράμματος

Η exec() αντικαθιστά την τρέχουσα διεργασία με ένα νέο πρόγραμμα — σαν "μεταμόρφωση": το PID παραμένει ίδιο, αλλά ο κώδικας, τα δεδομένα, το stack αντικαθίστανται πλήρως. Ο κώδικας μετά την exec() δεν εκτελείται ποτέ αν πετύχει. Τυπικά πρώτα κάνουμε fork() και μετά στο child exec() — έτσι ξεκινάμε νέα προγράμματα χωρίς να σκοτώσουμε τον parent.

Function Path Args Env Παράδειγμα κλήσης
execl() full path list inherit execl("/bin/ls","ls","-l",NULL)
execv() full path array inherit execv("/bin/ls", argv)
execlp() PATH ✓ list inherit execlp("ls","ls","-l",NULL)
execvp() PATH ✓ array inherit execvp("ls", argv)
execve() full path array explicit ✓ execve(path, argv, envp)
🧠
Tip — πώς να θυμάσαι ποια είναι ποια
Το suffix λέει τα πάντα:
l = list — arguments ως ξεχωριστά strings: execl(path, "ls", "-l", NULL)
v = vector/array — arguments ως array: char *argv[] = {"ls","-l",NULL}
p = PATH search — δεν χρειάζεσαι full path, βρίσκει αυτόματα στο PATH
e = environment — περνάς ρητά το environment (envp)
execvp = vector + PATH search = το πιο χρήσιμο σε shells
// τυπικό pattern: fork() + exec() pid_t pid = fork(); if (pid == 0) { // CHILD: αντικαθίσταται από ls execlp("ls", "ls", "-la", NULL); // αν φτάσω εδώ, η exec απέτυχε perror("exec"); exit(1); } else if (pid > 0) { waitpid(pid, NULL, 0); } else { perror("fork"); exit(1); }
// execv με array char *args[] = {"ls", "-la", NULL}; // ΠΡΟΣΟΧΗ: το τελευταίο ΠΑΝΤΑ NULL! execvp("ls", args); // exec με pipe — child γράφει στο pipe if (pid == 0) { dup2(pipe_fd[1], 1); // stdout → pipe close(pipe_fd[0]); close(pipe_fd[1]); execlp("ls", "ls", NULL); }
Τι κρατάει μετά exec()
Open file descriptors παραμένουν (εκτός αν έχουν FD_CLOEXEC). Αυτό επιτρέπει να περνάμε pipes στα εκτελούμενα προγράμματα μέσω dup2() πριν την exec().
⏳

wait() και waitpid() — Συλλογή Exit Status

Όταν ένας πατέρας δημιουργεί μια νέα διεργασία (child), πρέπει να φροντίσει για τον τερματισμό της. Η wait() επιτρέπει στον parent να συγχρονιστεί με το παιδί του: σταματά την εκτέλεση μέχρι το παιδί να τελειώσει, μαθαίνει αν η εργασία πέτυχε (exit status) και ελευθερώνει την εγγραφή του παιδιού από τον πίνακα διεργασιών (προλαμβάνοντας τα zombies).

#include <sys/wait.h> int status; pid_t dead_pid; // wait: περιμένει ΟΠΟΙΟΔΗΠΟΤΕ child dead_pid = wait(&status); // waitpid: περιμένει ΣΥΓΚΕΚΡΙΜΕΝΟ child dead_pid = waitpid(child_pid, &status, 0); // -1: περιμένει ΟΠΟΙΟΔΗΠΟΤΕ (ίδιο με wait) dead_pid = waitpid(-1, &status, 0); // WNOHANG: non-blocking, επιστρέφει 0 αν δεν τελείωσε dead_pid = waitpid(-1, &status, WNOHANG); // ανάλυση exit status if (WIFEXITED(status)) { printf("exit: %d\n", WEXITSTATUS(status)); } if (WIFSIGNALED(status)) { printf("killed by: %d\n", WTERMSIG(status)); }
🧠
Tip — macros από το όνομα
WIFEXITED = "W"ait "IF" child "EXITED" normally (exit/return)
WIFSIGNALED = "W"ait "IF" killed by "SIGNAL"
WEXITSTATUS = ποιος ήταν ο exit code (0-255)
WTERMSIG = ποιο signal το σκότωσε
Το W μπροστά = "Wait macro".
waitpid — πρώτο argument
// pid > 0: περιμένει ΑΥΤΟ το child waitpid(child_pid, &status, 0); // pid == -1: περιμένει ΟΠΟΙΟΔΗΠΟΤΕ child waitpid(-1, &status, 0); // = wait() // pid == 0: οποιοδήποτε child ΤΗΣ ΙΔΙΑΣ process group waitpid(0, &status, 0); // WNOHANG: αν δεν τελείωσε κανένα child, // επιστρέφει 0 αμέσως (μη-blocking) waitpid(-1, &status, WNOHANG);
🧠
Tip — WNOHANG
Wait NO HANGing = "περίμενε χωρίς να κολλήσεις". Χρήσιμο σε loops: ελέγχεις αν τελείωσε child χωρίς να μπλοκάρεις.

Zombie & Orphan Processes

Δύο ειδικές καταστάσεις προκύπτουν από λάθος διαχείριση του κύκλου ζωής μιας διεργασίας. Ο Zombie: το παιδί τελείωσε αλλά κανείς δεν "διάβασε" την κατάστασή του. Ο Orphan: ο γονέας τελείωσε πριν το παιδί. Τα δύο αυτά σενάρια έχουν εντελώς διαφορετικές συνέπειες.

🧟 Zombie Process

Child τερμάτισε αλλά ο parent δεν έχει κάνει wait() ακόμα. Ο kernel κρατά ζωντανή μια εγγραφή (exit status) στον process table — κανείς δεν την έχει "μαζέψει".

// zombie: parent "ξεχνά" να κάνει wait if (fork() == 0) { exit(0); // child τελειώνει αμέσως } // parent συνεχίζει χωρίς wait() sleep(30); // ps aux δείχνει [defunct] ή Z
🚨 Γιατί είναι κακό
Κάθε zombie καταλαμβάνει μια εγγραφή στον process table. Αν δεν "μαζεύεις" τα παιδιά σου, ο πίνακας εξαντλείται και δεν μπορείς να φτιάξεις νέες διεργασίες.
🧠
Λύση
Πάντα wait() ή waitpid() μετά από fork(). Για servers: handler για SIGCHLD που κάνει waitpid(-1, NULL, WNOHANG) σε loop.
👻 Orphan Process

Parent τερμάτισε πριν το child. Ο kernel υιοθετεί αυτόματα τα ορφανά παιδιά στον init (PID 1).

// orphan: parent πεθαίνει πρώτος if (fork() == 0) { sleep(30); // child τρέχει ακόμα exit(0); } exit(0); // parent εξάγεται πρώτος // child γίνεται orphan // → init (PID 1) γίνεται ο νέος parent
✅ Δεν είναι πρόβλημα
Τα orphans δεν είναι πρόβλημα — ο init κάνει wait() για αυτά. Δεν "συσσωρεύονται" γιατί ο init τα φροντίζει.
🧠
Tip — ορφανός vs zombie
Orphan = "γονέας πέθανε πρώτος" → υιοθετείται από init → ΟΚ.
Zombie = "παιδί πέθανε, γονέας δεν ρώτησε πώς" → μένει στον πίνακα → πρόβλημα.
PROCESS TREE — παράδειγμα
1 init/systemd
└── 1234 bash
    └── 1235 my_server (parent)
        ├── 1236 worker_1 (running OK)
        ├── 1237 worker_2 [Z] ← ZOMBIE — τελείωσε, parent δεν έκανε wait()
        └── 1238 worker_3 ← αν parent (1235) πεθάνει, γίνεται ORPHAN → υιοθετεί ο init

Anonymous Pipes — pipe()

Ένα pipe είναι ένα μονής κατεύθυνσης κανάλι επικοινωνίας στη μνήμη. Φαντάσου το σαν ένα σωλήνα όπου η μία διεργασία ρίχνει δεδομένα από τη μια άκρη και η άλλη τα μαζεύει από την άλλη. Είναι ο βασικός τρόπος για να "μιλήσουν" μεταξύ τους διεργασίες που έχουν κοινή καταγωγή (π.χ. parent-child).

int p[2]; if (pipe(p) == -1) { perror("pipe"); exit(1); } // p[0] = read end, p[1] = write end pid_t pid = fork(); if (pid == 0) { // CHILD: reader close(p[1]); // κλείνω write end char buf[256]; ssize_t n = read(p[0], buf, sizeof(buf)); buf[n] = '\0'; printf("got: %s\n", buf); close(p[0]); exit(0); } else { // PARENT: writer close(p[0]); // κλείνω read end write(p[1], "hello", 5); close(p[1]); // κλείνω → EOF για reader wait(NULL); }
PARENT
writer
CHILD
reader
p[1] write
→ data flows → → →
p[0] read
Parent: close(p[0]) · Child: close(p[1])
🧠
Tip — p[0] και p[1]
p[0] = read end. Θυμήσου: 0 = stdin = ανάγνωση.
p[1] = write end. Θυμήσου: 1 = stdout = εγγραφή.
Η σειρά στο array ακολουθεί την κατεύθυνση: 0→in, 1→out.
🚨 Γιατί κλείνουμε αχρείαστα ends;
Αν ο reader δεν κλείσει το write end (p[1]), ο ίδιος είναι "potential writer" οπότε δεν θα δει ποτέ EOF και θα μπλοκάρει στο read() για πάντα. EOF έρχεται μόνο όταν ΟΛΑ τα write ends έχουν κλείσει.
🧠
Tip — SIGPIPE
Αν γράψεις σε pipe που δεν έχει κανέναν reader (το read end έχει κλείσει), λαμβάνεις SIGPIPE (signal 13). "PIPE broken". Αν το αγνοήσεις, η write() επιστρέφει -1 με errno=EPIPE.
📡

Named Pipes (FIFOs) — mkfifo()

Τα FIFOs είναι pipes που έχουν όνομα στο filesystem. Ενώ τα anonymous pipes χάνονται όταν κλείσουν οι διεργασίες, ένα FIFO υπάρχει σαν αρχείο (αν και τα δεδομένα του δεν αποθηκεύονται στο δίσκο). Αυτό επιτρέπει σε εντελώς άσχετες διεργασίες να επικοινωνήσουν απλά ανοίγοντας το ίδιο όνομα αρχείου.

🧠
Tip — FIFO = First In First Out
Το "FIFO" δεν είναι τυχαίο — τα δεδομένα βγαίνουν στην ίδια σειρά που μπήκαν. Σαν ουρά αναμονής. Η λέξη "named" = έχει όνομα στο filesystem (αντίθεση με anonymous pipe).
#include <sys/stat.h> #include <fcntl.h> // 1. δημιουργία FIFO (μία φορά) if (mkfifo("myfifo", 0666) == -1) { if (errno != EEXIST) { // OK αν ήδη υπάρχει perror("mkfifo"); exit(1); } } // 2. WRITER process (ξεχωριστό πρόγραμμα) int fd = open("myfifo", O_WRONLY); // ↑ ΜΠΛΟΚΑΡΕΙ ώσπου reader να ανοίξει! write(fd, "hello", 5); close(fd); // 3. READER process (ξεχωριστό πρόγραμμα) fd = open("myfifo", O_RDONLY); // ↑ ΜΠΛΟΚΑΡΕΙ ώσπου writer να ανοίξει! read(fd, buf, sizeof(buf)); close(fd);
Blocking & Deadlock

Το default open() σε FIFO μπλοκάρει ώσπου και τα δύο άκρα ανοίξουν. Αυτό μπορεί να δημιουργήσει deadlock:

// DEADLOCK — κλασικό σενάριο: // Process A: open("fifo1", O_WRONLY) → μπλοκ // Process B: open("fifo2", O_WRONLY) → μπλοκ // Κανείς δεν ανοίγει το read end του άλλου! // ΛΥΣΗ: O_NONBLOCK fd = open("myfifo", O_WRONLY|O_NONBLOCK); // επιστρέφει -1 (ENXIO) αν δεν υπάρχει reader // αντί να μπλοκάρει — ο caller αποφασίζει
🧠
Tip — ENXIO
ENXIO = "No Such Device Or Address". Με O_NONBLOCK σε FIFO: "δεν υπάρχει reader ακόμα". Αυτό σου επιτρέπει να προσπαθήσεις ξανά αργότερα.

Σημαντικά Patterns — Εξετάσεις

Εδώ βλέπουμε πώς συνδυάζονται στην πράξη όλες οι παραπάνω έννοιες (fork, exec, pipe, dup2). Αυτά τα σχήματα αποτελούν τη βάση για τη λειτουργία ενός Shell και είναι από τα πιο συνηθισμένα θέματα στις εξετάσεις.

Shell Pipeline: ls | wc -l
int p[2]; pipe(p); // child 1: εκτελεί "ls", γράφει στο pipe if (fork() == 0) { close(p[0]); dup2(p[1], 1); // stdout → pipe write close(p[1]); execlp("ls", "ls", NULL); perror("exec ls"); exit(1); } // child 2: εκτελεί "wc -l", διαβάζει από pipe if (fork() == 0) { close(p[1]); dup2(p[0], 0); // stdin ← pipe read close(p[0]); execlp("wc", "wc", "-l", NULL); perror("exec wc"); exit(1); } // parent κλείνει ΚΑΙ τα δύο ends close(p[0]); close(p[1]); wait(NULL); wait(NULL); // 2 children
🧠
Tip — η σειρά των βημάτων
Σε κάθε child που χρησιμοποιεί pipe:
1. close αχρείαστο end
2. dup2 για σύνδεση stdin/stdout
3. close αρχικό pipe fd
4. exec
Ο parent κλείνει και τα δύο pipe ends μετά τα forks.
SIGCHLD Handler — Zombie Prevention
void sigchld_handler(int sig) { int status; pid_t pid; // loop μέχρι να μαζέψουμε ΟΛΑ // τα παιδιά που τελείωσαν while ((pid = waitpid(-1, &status, WNOHANG)) > 0) { /* zombie ελευθερώθηκε */ } } // εγκατάσταση handler struct sigaction sa; sa.sa_handler = sigchld_handler; sa.sa_flags = SA_RESTART; sigemptyset(&sa.sa_mask); sigaction(SIGCHLD, &sa, NULL);
🧠
Tip — SA_RESTART
SA_RESTART = "μετά τον handler, επανέναρξε αυτόματα τυχόν interrupted system calls". Χωρίς αυτό, system calls (read, accept…) επιστρέφουν -1 με errno=EINTR όταν έρθει signal.
